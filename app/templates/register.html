from flask_wtf import FlaskForm from wtforms import StringField from wtforms.validators import DataRequired, Length class RegistrationForm(FlaskForm): username = StringField('Username', validators=[DataRequired(), Length(min=10, max=255)]) email = StringField('Email', validators=[DataRequired(), Email()]) password = PasswordField('Password', validators=[DataRequired(), Length(min=10, max=255)]) confirm_password = PasswordField('Confirm Password', validators=[DataRequired(), EqualTo('password')]) submit = SubmitField('Sign Up')
